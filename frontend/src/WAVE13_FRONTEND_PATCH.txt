const getBundleMeta = (section) => {
  const raw = section?.raw_row || {};
  const linkedSections = raw.linked_sections || section?.linked_sections || [];
  const distinctCrns = new Set(
    [
      ...(section?.primary_crn ? String(section.primary_crn).split("/").map((x) => x.trim()) : []),
      ...linkedSections
        .map((s) => String(s?.primary_crn || "").trim())
        .filter(Boolean),
    ].filter(Boolean)
  );

  const bundleType = raw.bundle_type || section?.bundle_type || "single";
  const isTrueLinkedBundle =
    raw.is_true_linked_bundle ??
    section?.is_true_linked_bundle ??
    (
      distinctCrns.size > 1 &&
      ["cross_listed", "corequisite_bundle", "mixed_bundle"].includes(bundleType)
    );

  return {
    linkedSections,
    distinctCrnCount: raw.distinct_crn_count || section?.distinct_crn_count || distinctCrns.size,
    bundleType,
    isTrueLinkedBundle,
  };
};
